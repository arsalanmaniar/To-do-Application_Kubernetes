1. Get the application URLs by running these commands:

{{- if contains "NodePort" .Values.frontend.service.type }}
  export FRONTEND_NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "todo-chatbot.fullname" . }}-frontend-service)
  export FRONTEND_NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo http://$FRONTEND_NODE_IP:$FRONTEND_NODE_PORT
{{- else if contains "ClusterIP" .Values.frontend.service.type }}
  export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "todo-chatbot.name" . }},app.kubernetes.io/instance={{ .Release.Name }}" -o jsonpath="{.items[0].metadata.name}")
  echo "Visit http://127.0.0.1:8080 to use your application"
  kubectl --namespace {{ .Release.Namespace }} port-forward $POD_NAME 8080:3000
{{- end }}

{{- if contains "NodePort" .Values.backend.service.type }}
  export BACKEND_NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "todo-chatbot.fullname" . }}-backend-service)
  export BACKEND_NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo http://$BACKEND_NODE_IP:$BACKEND_NODE_PORT
{{- else if contains "ClusterIP" .Values.backend.service.type }}
  export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "todo-chatbot.name" . }},app.kubernetes.io/instance={{ .Release.Name }}" -o jsonpath="{.items[0].metadata.name}")
  echo "Visit http://127.0.0.1:8081 to use your backend"
  kubectl --namespace {{ .Release.Namespace }} port-forward $POD_NAME 8081:8000
{{- end }}

2. To check the status of pods:
  kubectl get pods --namespace {{ .Release.Namespace }}

3. To view logs:
  kubectl logs deployment/{{ include "todo-chatbot.fullname" . }}-frontend --namespace {{ .Release.Namespace }}
  kubectl logs deployment/{{ include "todo-chatbot.fullname" . }}-backend --namespace {{ .Release.Namespace }}
